pipeline {
    agent any
    tools {
        maven 'M3'
        jdk 'JDK11'
    }
    stages {
        stage('Checkout') {
            steps {
                git 'https://github.com/Dasarivamsil/my-maven-app-deployment.git'
            }
        }
        stage('Build & Test') {
            steps {
                sh 'mvn clean compile test'
            }
        }
        stage('Package') {
            steps {
                sh 'mvn package -DskipTests'
                archiveArtifacts 'target/*.jar'
            }
        }
    }
    post {
        always {
            echo 'Build completed!'
        }
    }
}
